namespace MyNamespace
{
    public static partial class ColourExtensions
    {
        public static string ToStringFast(this MyNamespace.Colour value)
        {
            return value switch
            {
                MyNamespace.Colour.Red => nameof(MyNamespace.Colour.Red),
                MyNamespace.Colour.Blue => nameof(MyNamespace.Colour.Blue),
                _ => value.ToString()
            };
        }
    }
}
