namespace MyNamespace
{
    public static partial class EnumExtensions
    {
        public static string ToStringFast(this MyNamespace.Gender value)
        {
            return value switch
            {
                MyNamespace.Gender.None => nameof(MyNamespace.Gender.None),
                MyNamespace.Gender.Male => nameof(MyNamespace.Gender.Male),
                MyNamespace.Gender.Female => nameof(MyNamespace.Gender.Female),
                _ => value.ToString()
            };
        }
    }
}
